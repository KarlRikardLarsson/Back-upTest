///$tab **Config**
TRACE Initializing configuration and connection settings;

SET vu_base_store_space_name='D1 - Extract';

// Connection to store and retrieve QVD files
SET vu_qvd_storage_connection = '$(vu_base_store_space_name):DataFiles';

// Output Store
SET vOutputQVDName = 'lib://$(vu_qvd_storage_connection)/Riksbanken/Valutor_Riksbanken.qvd';
SET vInputQVDName    = 'lib://$(vu_qvd_storage_connection)/Fortnox/*/Currencies.qvd';

// Connection used to retrieve actual data (Currency exchange + date)
SET vu_rest_riksbanken = '$(vu_base_store_space_name):Riksbanken_ExchangeRates';

// Rest Base URL for extracting data
LET vu_api_base_url      = 'https://api.riksbank.se/swea/v1/Observations/$1/$2';

///$tab Get currencies
// Load currency data from QVD
Currencies:
LOAD
    Code,
    StartDate
FROM '$(vInputQVDName)'
(qvd);

// Get the minimum available date from the currency QVD
MinDate:
LOAD 
    Min(StartDate) AS MinDate
RESIDENT Currencies;

LET vWeekendSafe           = if(match(weekday(Peek('MinDate', 0, 'MinDate')),'lör','sön'),3,0);

// Substract 3 days from start date. In case first date is on a sunday the peek does not work and cause the first date not to have a value
LET vStartDate           = Date(Peek('MinDate', 0, 'MinDate')-vWeekendSafe, 'YYYY-MM-DD');
LET vStartDateNumerical  = Num(Peek('MinDate', 0, 'MinDate')-vWeekendSafe);

DROP TABLE MinDate;

// Create a distinct list of currency codes
Currencies_of_interest:
LOAD DISTINCT 
    Code AS CurrencyCodes
RESIDENT Currencies;

DROP TABLE Currencies;

// ----------------- Fetch exchange rate series info ----------------- //
// Connect to Riksbank API (Series metadata)
LIB CONNECT TO '$(vu_rest_riksbanken)';

RestConnectorMasterTable:
SQL SELECT 
    "seriesId",
    "source",
    "shortDescription",
    "midDescription",
    "longDescription",
    "groupId",
    "observationMaxDate",
    "observationMinDate",
    "seriesClosed"
FROM JSON (wrap on) "root";

ExchangeRateSeries:
LOAD
    seriesId,
    source,
    shortDescription,
    midDescription,
    longDescription,
    groupId,
    observationMaxDate,
    observationMinDate,
    seriesClosed
RESIDENT RestConnectorMasterTable;

DROP TABLE RestConnectorMasterTable;

// Initialize target table
Currency:
LOAD * INLINE [
%DateKey
]

WHERE 1=0; // Empty structure, used for appending later

FOR EACH vCurrency IN FieldValueList('CurrencyCodes')

    TRACE Fetching exchange rate for currency: $(vCurrency);

    GetCurrencySeries:
    LOAD
        seriesId
    RESIDENT ExchangeRateSeries
    WHERE shortDescription = '$(vCurrency)';

    // Check if seriesId was found
    IF NoOfRows('GetCurrencySeries') > 0 THEN

        LET vSeriesId = Peek('seriesId', 0, 'GetCurrencySeries');
        DROP TABLE GetCurrencySeries;

        TRACE Found seriesId: $(vSeriesId);

        LET vURL = '$(vu_api_base_url($(vSeriesId),$(vStartDate)))';

        RestConnectorMasterTable:
        SQL SELECT 
            "date",
            "value"
        FROM JSON (wrap on) "root"
        WITH CONNECTION (
            URL "$(vURL)"
        );

        LET vEndDate = Num(Today());
        
        Currencies_tmp:
        LOAD 
            '$(vCurrency)' AS Currency,
            Date($(vStartDateNumerical) + IterNo() - 1, 'YYYY-MM-DD') AS %DateKey
        AUTOGENERATE 1
        WHILE $(vStartDateNumerical) + IterNo() - 1 <= $(vEndDate);
        

        LEFT JOIN (Currencies_tmp)
        LOAD
            Date(Floor(date)) AS %DateKey,
            value
        RESIDENT RestConnectorMasterTable;

        DROP TABLE RestConnectorMasterTable;
		
        NoConcatenate
        Currencies_tmp2:
        LOAD
            Currency,
            %DateKey,
			Coalesce(value, Peek(value, -1), Peek(CurrencyFactor, -1)) AS CurrencyFactor            
        RESIDENT Currencies_tmp
        ORDER BY %DateKey;
        
        DROP TABLE Currencies_tmp;

		CONCATENATE(Currency)
        LOAD
            %DateKey,
            Currency,
            CurrencyFactor
        RESIDENT Currencies_tmp2;

        DROP TABLE Currencies_tmp2;
        
    ELSE
        TRACE Skipping: No seriesId found for $(vCurrency);
        DROP TABLE GetCurrencySeries;
    ENDIF

NEXT


// Final cleanup
DROP TABLES ExchangeRateSeries, Currencies_of_interest;

// Store the final currency data
STORE Currency INTO [$(vOutputQVDName)] (qvd);
