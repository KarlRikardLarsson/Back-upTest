///$tab CheckTokenExpired
// Subroutine: Checks if the access token has expired
SUB CheckTokenExpired

    // Access tokens last 60 minutes – we'll refresh after 58 minutes
    LET vTokenExpireLimit = 58 * 60; // seconds
    
	SavedTimestamp:
	LOAD SavedTimestamp,AccessToken FROM '$(vRefreshTokenQVDName)' 
	(qvd)
	WHERE Company ='$(vCompany)';
	
    LET vSavedTimestamp =Peek('SavedTimestamp', 0, 'SavedTimestamp');
    
	// Read current time and token expiry
	LET vNow = Timestamp(ConvertToLocalTime(now(1),'$(vGMToffset)')); 

	// Calculate difference (in minutes)
	LET vMinutesLeft = Round((alt(vSavedTimestamp,vNow) - vNow) * 24 * 60); // If null, vMinutesLeft = now()


	// Trace current situation
	TRACE 🕑 Token Expires In: $(vMinutesLeft) minutes (Margin: $(vTokenExpiryMarginMinutes) minutes);
	
	// Decision logic
	IF vMinutesLeft <= vTokenExpiryMarginMinutes THEN
	
	    TRACE 🔄 Token is about to expire (or expired)! Requesting new token...;
	        
	        DISCONNECT;
	
	        CALL UpdateTokens('$(vCompany)');
	
	        LIB CONNECT TO '$(vConnection)';
	ELSE
	    TRACE ✅ Current token is still valid. No need to refresh.;
    	LET vAccessToken =Peek('AccessToken', 0, 'SavedTimestamp');
        
	ENDIF
    
	DROP TABLE SavedTimestamp;

END SUB

///$tab UpdateTokens
// Updates the refresh token and retrieves a new access token for the specified company
SUB UpdateTokens(vCompany)

    Trace Updating tokens for company $(vCompany);
    LET qvd_name = '$(vRefreshTokenQVDName)';

    // Load existing refresh tokens from QVD
    Refreshtokens_full:
    NoConcatenate
    LOAD 
        Company,
        RefreshToken,
        AccessToken,
        SavedTimestamp
    FROM '$(qvd_name)'(qvd);
	
    // Get the refresh token for the specified company
    LET vInputRefreshToken = lookup('RefreshToken', 'Company', '$(vCompany)', 'Refreshtokens_full');

    TRACE Making token request using refresh token: $(vInputRefreshToken);

    LIB CONNECT TO '$(vRefreshConnection)';

    // Send token refresh request to Fortnox API
    RestConnectorMasterTable:
    SQL SELECT 
        "refresh_token",
        "access_token",
        "__KEY_root"
    FROM JSON (wrap on) "root" PK "__KEY_root"
    WITH CONNECTION (
        BODY "grant_type=refresh_token&refresh_token=$(vInputRefreshToken)"
    );

    
    // Extract new tokens from response
    LET vOutputRefreshToken = peek('refresh_token', 0, 'RestConnectorMasterTable');
    LET vAccessToken        = peek('access_token', 0, 'RestConnectorMasterTable');

    DROP TABLE RestConnectorMasterTable;


    
    RefreshToken:
    NoConcatenate
    LOAD 
        '$(vCompany)' as Company, 
        '$(vOutputRefreshToken)' AS RefreshToken,
        '$(vAccessToken)' 		AS AccessToken,
         ConvertToLocalTime(now(1) + 1/24, '$(vGMToffset)') AS SavedTimestamp // adding one hour to get exipre time
    AutoGenerate (1)
	;
    
    // Create updated entry for the current company
    Concatenate(RefreshToken)
    Load *
    RESIDENT Refreshtokens_full
    WHERE Company <> '$(vCompany)';

    DROP TABLE Refreshtokens_full;

    // Save updated token list back to QVD
    Store RefreshToken INTO '$(qvd_name)' (qvd);
	DROP TABLE RefreshToken;
    
    TRACE Disconnecting from token refresh service;
    DISCONNECT;
    
END SUB



///$tab Financial Years
SUB Extract_FinancialYears

    TRACE Extracting Financial Years for $(vCompany);       
    
    // Set up
	SET vURL = '$(vu_api_base_url(financialyears))';
    LET vTable = 'FinancialYearInfo';

	LIB CONNECT TO '$(vConnection)';
    
    // Fetch Financial Years data from Fortnox API
    RestConnectorMasterTable:
    SQL SELECT 
        "@url",
        "FromDate",
        "ToDate",
        "Id"
    FROM JSON (wrap off) "FinancialYears" PK "__KEY_FinancialYears"
    WITH CONNECTION (
        URL "$(vURL)",
        HTTPHEADER "Authorization" "Bearer $(vAccessToken)"
    );

	// Filter and structure the data
    [$(vTable)]:
    LOAD	
        '$(vCompany)' AS Company,
        [Id],
        [FromDate],		
        [ToDate]
    RESIDENT RestConnectorMasterTable
    WHERE left(FromDate, 4) >= ($(vDefaultStartYear))
    Order by Id ASC; 

	// Clean up
    DROP TABLE RestConnectorMasterTable;
    
    // --- Build variable list of all Financial Year IDs ---
	LET vAllFinancialYears = '';
	FOR Fy = 0 TO NoOfRows('$(vTable)') - 1
	    LET vThisYear = Peek('Id', $(Fy), '$(vTable)');
	    LET vAllFinancialYears = if(len(trim('$(vAllFinancialYears)')) = 0, chr(39) & vThisYear & chr(39), '$(vAllFinancialYears)' & ',' & chr(39) & vThisYear & chr(39));
	NEXT Fy

	// Store to QVD
    CALL store_qvd('$(vTable)');

END SUB // Extract.FinancialYears

///$tab VoucherList
SUB Extract_Vouchers

    TRACE Starting Extract_Vouchers for $(vCompany);

    // --- Setup ---
    SET vURL = '$(vu_api_base_url(vouchers))';
    LET vNoOfPages   = 0;
    LET vCounter     = 0;
    LET vTable       = 'Vouchers';

    // --- Check for existing QVD and determine incremental load date ---
    CALL Check_incremental('$(vTable)');

    LIB CONNECT TO '$(vConnection)';

    // -------------------------------------------------------------------------
    // Load remaining URLs from previous failed run (if incremental)
    // -------------------------------------------------------------------------
    IF $(vIncrementalQVDsAvailable) = 1 THEN
        LET qvd_url_name = '$(vOutputQVDName($(vCompany),Incremental_Settings/Remove_Voucher_URLs))';

        tmp_Voucher_URLs:
        LOAD DISTINCT *
        FROM '$(qvd_url_name)' (qvd);

        Voucher_URL:
        NoConcatenate
        LOAD *
        RESIDENT tmp_Voucher_URLs
        ORDER BY URL ASC;

        DROP TABLE tmp_Voucher_URLs;
    END IF

    // -------------------------------------------------------------------------
    // Loop through financial years and fetch voucher URLs
    // -------------------------------------------------------------------------
    FOR EACH vFinancialYearID IN $(vAllFinancialYears)

        TRACE Processing Financial Year: $(vFinancialYearID);

        // --- Get total pages and resources from API meta ---
        tmpMetaInfoVouchers:
        SQL SELECT 
            "__KEY_root",
            (SELECT 
                "@TotalPages",
                "@TotalResources",
                "__FK_MetaInformation"
             FROM "MetaInformation" FK "__FK_MetaInformation")
        FROM JSON (wrap on) "root" PK "__KEY_root"
        WITH CONNECTION (
            URL "$(vURL)?limit=$(vAPILimit)&financialyear=$(vFinancialYearID)&lastmodified=$(vIncrementalStartTime)",
            HTTPHEADER "Authorization" "Bearer $(vAccessToken)"
        );

        MetaInfoVouchers:
        LOAD
            [@TotalPages]     AS MaxPages,
            [@TotalResources] AS TotalResources
        RESIDENT tmpMetaInfoVouchers
        WHERE NOT IsNull([__FK_MetaInformation]);

        LET vNoOfPages      = alt(Peek('MaxPages',      0, 'MetaInfoVouchers'), 0);
        LET vTotalResources = alt(Peek('TotalResources', 0, 'MetaInfoVouchers'), 0);

        LET vTraceMessage = If($(vTotalResources) > 0,
            'Total Voucher pages to load for FY $(vFinancialYearID): $(vNoOfPages)',
            'No updated records for FY $(vFinancialYearID). Skipping.');
        TRACE $(vTraceMessage);
        DROP VARIABLE vTraceMessage;
        DROP TABLE MetaInfoVouchers, tmpMetaInfoVouchers;

        // --- Fetch all voucher URLs for this financial year (paginated) ---
        FOR vPage = 1 TO $(vNoOfPages)

            TRACE Loading Vouchers - FY: $(vFinancialYearID), Page $(vPage) of $(vNoOfPages);

            RestConnectorMasterTable:
            SQL SELECT 
                "__KEY_root",
                (SELECT "@url" FROM "Vouchers" FK "__FK_Vouchers")
            FROM JSON (wrap on) "root" PK "__KEY_root"
            WITH CONNECTION (
                URL "$(vURL)?limit=$(vAPILimit)&financialyear=$(vFinancialYearID)&lastmodified=$(vIncrementalStartTime)&page=$(vPage)",
                HTTPHEADER "Authorization" "Bearer $(vAccessToken)"
            );

            Voucher_URL:
            LOAD DISTINCT [@url] AS URL
            RESIDENT RestConnectorMasterTable
            WHERE Len([@url]) > 0;

            DROP TABLE RestConnectorMasterTable;

        NEXT vPage

    NEXT vFinancialYearID

    // --- Merge with incremental URLs if applicable ---
    IF $(vIncrementalQVDsAvailable) = 1 THEN
        CALL Incremental_Load('Incremental_Settings/Remove_Voucher_URLs');
    END IF

    // --- Exit if nothing to process ---
    EXIT SUB UNLESS NoOfRows('Voucher_URL') > 0;

    // -------------------------------------------------------------------------
    // Loop through each voucher URL and extract details
    // -------------------------------------------------------------------------
    FOR EACH vLoopUrl IN FieldValueList('URL')

        TRACE Fetching Voucher from URL: $(vLoopUrl);
        CALL Extract_VoucherDetails('$(vLoopUrl)');
        CALL CheckTokenExpired;

        // --- Remove processed URL and update resume QVD ---
        Remove_Voucher_URLs:
        NoConcatenate
        LOAD DISTINCT URL
        RESIDENT Voucher_URL
        WHERE URL <> '$(vLoopUrl)'
        ORDER BY URL ASC;

        DROP TABLE Voucher_URL;

        LET qvd_url_name = '$(vOutputQVDName($(vCompany),Incremental_Settings/Remove_Voucher_URLs))';
        TRACE Storing resume QVD to: $(qvd_url_name);
        STORE Remove_Voucher_URLs INTO '$(qvd_url_name)' (qvd);

        RENAME TABLE Remove_Voucher_URLs TO Voucher_URL;

        // --- Append company metadata and store voucher data ---
        JOIN($(vTable)_Rows)
        LOAD
            *,
            '$(vStartTime)' AS lastmodify,
            '$(vCompany)'   AS Company
        RESIDENT [$(vTable)];

        DROP TABLE [$(vTable)];
        RENAME TABLE [$(vTable)_Rows] TO [$(vTable)];

        // --- Store incremental QVD ---
        CALL Incremental_Load('$(vTable)');
        CALL store_qvd('$(vTable)');
        exit script

    NEXT vLoopUrl

    DROP TABLE Voucher_URL;

    // -------------------------------------------------------------------------
    // Final processing
    // -------------------------------------------------------------------------
    CALL Incremental_Load('$(vTable)');
    CALL store_qvd('$(vTable)');

END SUB

              "TransactionInformation",

              "__FK_VoucherRows"

          WITH CONNECTION (
          URL "$(_URL)",

        [CostCenter],
        [Credit],

        [TransactionInformation],
        [Quantity],
        [__FK_VoucherRows] & '|' & $(vCounter) & '|' & '$(vUniqueKey)' AS [__KEY_Voucher]

    WHERE NOT IsNull([__FK_VoucherRows]);

          [ApprovalState],
          [__KEY_Voucher] & '|' & $(vCounter) & '|' & '$(vUniqueKey)' AS [__KEY_Voucher]

      WHERE NOT IsNull([__KEY_Voucher]);

      DROP TABLE RestConnectorMasterTable;

	  LET vCounter = $(vCounter) + 1;

SUB Extract_SupplierInvoices

    TRACE Starting Extract_SupplierInvoices for $(vCompany);

    // --- Setup ---
    SET vURL = '$(vu_api_base_url(supplierinvoices))';
    LET vNoOfPages   = 0;
    LET vCounter     = 0;
    LET vTable       = 'SupplierInvoices';

    // --- Check for existing QVD and determine incremental load date ---
    CALL Check_incremental('$(vTable)');

    LIB CONNECT TO '$(vConnection)';

    // -------------------------------------------------------------------------
    // Get total pages and resources from API meta
    // -------------------------------------------------------------------------
    tmpMetaInfoSupplierInvoices:
    SQL SELECT 
        "__KEY_root",
        (SELECT 
            "@TotalPages",
            "@TotalResources",
            "__FK_MetaInformation"
         FROM "MetaInformation" FK "__FK_MetaInformation")
    FROM JSON (wrap on) "root" PK "__KEY_root"
    WITH CONNECTION (
        URL "$(vURL)?limit=$(vAPILimit)&lastmodified=$(vIncrementalStartTime)",
        HTTPHEADER "Authorization" "Bearer $(vAccessToken)"
    );

    MetaInfoSupplierInvoices:
    LOAD
        [@TotalPages]     AS MaxPages,
        [@TotalResources] AS TotalResources
    RESIDENT tmpMetaInfoSupplierInvoices
    WHERE NOT IsNull([__FK_MetaInformation]);

    LET vNoOfPages      = alt(Peek('MaxPages',      0, 'MetaInfoSupplierInvoices'), 0);
    LET vTotalResources = alt(Peek('TotalResources', 0, 'MetaInfoSupplierInvoices'), 0);

    LET vTraceMessage = If($(vTotalResources) > 0,
        'Total SupplierInvoice pages to load: $(vNoOfPages)',
        'No updated records since last reload.');
    TRACE $(vTraceMessage);
    DROP VARIABLE vTraceMessage;
    DROP TABLE MetaInfoSupplierInvoices, tmpMetaInfoSupplierInvoices;

    // -------------------------------------------------------------------------
    // Load remaining URLs from previous failed run (if incremental)
    // -------------------------------------------------------------------------
    IF $(vIncrementalQVDsAvailable) = 1 THEN
        LET qvd_url_name = '$(vOutputQVDName($(vCompany),Incremental_Settings/Remove_SupplierInvoice_URLs))';

        tmp_SupplierInvoice_URLs:
        LOAD DISTINCT *
        FROM '$(qvd_url_name)' (qvd);

        SupplierInvoice_URLs:
        NoConcatenate
        LOAD *
        RESIDENT tmp_SupplierInvoice_URLs
        ORDER BY URL ASC;

        DROP TABLE tmp_SupplierInvoice_URLs;
    END IF

    // --- Exit if nothing to process ---
    EXIT SUB UNLESS $(vTotalResources) > 0 OR NoOfRows('SupplierInvoice_URLs') > 0;

    // -------------------------------------------------------------------------
    // Fetch all supplier invoice URLs from API (paginated)
    // -------------------------------------------------------------------------
    FOR vPage = 1 TO $(vNoOfPages)

        TRACE Loading SupplierInvoices - Page $(vPage) of $(vNoOfPages);

        RestConnectorMasterTable:
        SQL SELECT 
            "__KEY_root",
            (SELECT "@url" FROM "SupplierInvoices" FK "__FK_SupplierInvoices")
        FROM JSON (wrap on) "root" PK "__KEY_root"
        WITH CONNECTION (
            URL "$(vURL)?limit=$(vAPILimit)&lastmodified=$(vIncrementalStartTime)&page=$(vPage)",
            HTTPHEADER "Authorization" "Bearer $(vAccessToken)"
        );

        SupplierInvoice_URLs:
        LOAD DISTINCT [@url] AS URL
        RESIDENT RestConnectorMasterTable
        WHERE Len([@url]) > 0;

        DROP TABLE RestConnectorMasterTable;

    NEXT vPage

    // --- Merge with incremental URLs if applicable ---
    IF $(vIncrementalQVDsAvailable) = 1 THEN
        CALL Incremental_Load('Incremental_Settings/Remove_SupplierInvoice_URLs');
    END IF

    // -------------------------------------------------------------------------
    // Loop through each supplier invoice URL and extract details
    // -------------------------------------------------------------------------
    FOR EACH vLoopUrl IN FieldValueList('URL')

        TRACE Fetching SupplierInvoice from URL: $(vLoopUrl);
        CALL Extract_SupplierInvoiceRows('$(vLoopUrl)');
        CALL CheckTokenExpired;

        // --- Remove processed URL and update resume QVD ---
        Remove_SupplierInvoice_URLs:
        NoConcatenate
        LOAD DISTINCT URL
        RESIDENT SupplierInvoice_URLs
        WHERE URL <> '$(vLoopUrl)'
        ORDER BY URL ASC;

        DROP TABLE SupplierInvoice_URLs;

        LET qvd_url_name = '$(vOutputQVDName($(vCompany),Incremental_Settings/Remove_SupplierInvoice_URLs))';
        TRACE Storing resume QVD to: $(qvd_url_name);
        STORE Remove_SupplierInvoice_URLs INTO '$(qvd_url_name)' (qvd);

        RENAME TABLE Remove_SupplierInvoice_URLs TO SupplierInvoice_URLs;

        // --- Append company metadata and store supplier invoice data ---
        JOIN($(vTable)_Rows)
        LOAD
            *,
            '$(vStartTime)' AS lastmodify,
            '$(vCompany)'   AS Company
        RESIDENT [$(vTable)];

        DROP TABLE [$(vTable)];
        RENAME TABLE [$(vTable)_Rows] TO [$(vTable)];

        // --- Store incremental QVD ---
        CALL Incremental_Load('$(vTable)');
        CALL store_qvd('$(vTable)');
        
        exit script

    NEXT vLoopUrl

END SUB

LET vIncrementalMessage          = If($(vIncrementalQVDsAvailable) = 1, 'Incremental QVDs found', 'No Incremental QVDs found. Do not panic. I can fix that.');


// ─────────────────────────────────────────────────────────────
// Determine Incremental Load Start Time
// ─────────────────────────────────────────────────────────────
IF $(vIncrementalQVDsAvailable) = 1 THEN

    [$(incrTable)MaxLastupdateTime]:
    LOAD
        Max(lastmodify) AS $(incrTable)MaxLastupdateTime
    FROM '$(vIncrQVDPath)' (qvd);

    LET vIncrementalStartTime = Timestamp(Peek('$(incrTable)MaxLastupdateTime', 0, '$(incrTable)MaxLastupdateTime') - (1 / 24 / 60), 'YYYY-MM-DD hh:mm'); //removing 1 hour 
    DROP TABLE [$(incrTable)MaxLastupdateTime];

    LET vIncrementalStartTimeDisplay = Timestamp('$(vIncrementalStartTime)', 'YYYY-MM-DD hh:mm:ss.fff');
    TRACE $(vIncrementalMessage) – resuming from: $(vIncrementalStartTimeDisplay);
    SET vIncrementalStartTimeDisplay =;

ELSE

    TRACE $(vIncrementalMessage) – full reload logic will be used.;
    LET vIncrementalStartTime = Timestamp(YearStart(Today(1) - vu_initial_days_back), 'YYYY-MM-DD hh:mm');
    
ENDIF

END SUB


SUB Incremental_Load(_Table)

    // Resolve QVD file path for the target table
    LET qvd_name = '$(vOutputQVDName($(vCompany),$(_Table)))';

    // Proceed if the QVD exists (incremental load is possible)
    IF $(vIncrementalQVDsAvailable) = 1 AND vForceFullReload <> 1 and NoOfRows('$(_Table)')>0 THEN

        TRACE Performing incremental merge for [$(_Table)] using QVD: $(qvd_name);

        // Build a map of new GuIDs to avoid duplicates
        Map_$(_Table)_NewUrl:
        MAPPING
        LOAD DISTINCT 
            [@url],
            1
        RESIDENT [$(_Table)];

        Concatenate($(_Table))
        LOAD *
        FROM [$(qvd_name)] (qvd)
        WHERE ApplyMap('Map_$(_Table)_NewUrl', [@url])<>1
        AND Date(floor(Date)) >= Date(floor('$(vRollingAnlysisDate)'));
        
        DROP MAPPING TABLE Map_$(_Table)_NewUrl;

    ELSE

        TRACE No previous QVD or new records found for [$(_Table)]. Skipping incremental merge.;

    END IF

END SUB
///$tab Store QVD
SUB store_qvd(_table)

    TRACE Starting QVD store for table: $(_table), company: $(vCompany);
    
    IF '$(_table)' = 'Inventory' THEN   //#003
    
    	    // --- Build QVD path dynamically ---
    	    LET qvd_name = '$(vOutputQVDName($(vCompany),$(_table)))';
	
			LET vIncrementalQVDsAvailable    = If(Len(FileTime('$(qvd_name)')) > 0, 1, 0);
    	    
    	    IF $(vIncrementalQVDsAvailable) = 1 THEN
    	    	
    	        // Building inventory history. Not loading todays
    	        Concatenate($(_table))
    	        LOAD * 
    	        FROM '$(qvd_name)' (qvd)
    	        WHERE Date(floor(lastmodify)) <> Date(Floor('$(vStartTime)'))
    	        AND Date(floor(lastmodify)) >= Date(floor('$(vRollingAnlysisDate)'));
    	    
    	    END IF
    	    
    	    TRACE Storing QVD to: $(qvd_name);
    	    STORE $(_table) INTO '$(qvd_name)' (qvd);
            
            // --- Log file metadata ---
    	    QVD_Metadata:
    	    LOAD 
    	        '$(vCompany)' AS Company,
    	        '$(_table)' AS Table,
    	        FileSize('$(qvd_name)') AS FileSize,
    	        FileTime('$(qvd_name)') AS FileTime,
    	        QvdNoOfFields('$(qvd_name)') AS FieldCount,
    	        QvdNoOfRecords('$(qvd_name)') AS RecordCount
    	    AUTOGENERATE 1;
            
            TRACE Dropping table $(_table);
    	    DROP TABLE [$(_table)];
    	    
	Else // if not inventory
    
    	IF NoOfRows('$(_table)') > 0  THEN
    	
    	        // --- Build QVD path dynamically ---
    	    LET qvd_name = '$(vOutputQVDName($(vCompany),$(_table)))';
	
    	    TRACE Storing QVD to: $(qvd_name);
    	    STORE $(_table) INTO '$(qvd_name)' (qvd);
	
    	    // --- Log file metadata ---
    	    QVD_Metadata:
    	    LOAD 
    	        '$(vCompany)' AS Company,
    	        '$(_table)' AS Table,
    	        FileSize('$(qvd_name)') AS FileSize,
    	        FileTime('$(qvd_name)') AS FileTime,
    	        QvdNoOfFields('$(qvd_name)') AS FieldCount,
    	        QvdNoOfRecords('$(qvd_name)') AS RecordCount
    	    AUTOGENERATE 1;
    	    
    	    TRACE Dropping table $(_table);
    	    DROP TABLE [$(_table)];
    	ELSE
    	
			TRACE No rows found in $(_table) for company $(vCompany). Nothing stored;
    	    
    	END IF
        
    END IF // If inventory

    TRACE Finished QVD store for $(qvd_name);

END SUB

///$tab Sheet timer
/*
---------------------------------------------------------------------------------------------------------------------------------------
Call this sub in the beginning and end of every section
to know how long each section takes to load
---------------------------------------------------------------------------------------------------------------------------------------
*/

// Start time with message
SUB StartTimer(timerName)
	TRACE ▶️ $(timerName);
    LET "vStart_$(timerName)" = now();
END Sub

// End time and calculate exection time
SUB EndTimer(timerName) 
	LET "vEnd_$(timerName)" = now(); 
    LET "vDuration_$(timerName)" = Interval(vEnd_$(timerName)-vStart_$(timerName),'hh:mm:ss');
    TRACE ⏱️ Execution time ($(timerName)): $(vDuration_$(timerName));
    
    TimerLog: 
    LOAD 
    	'$(timerName)'				AS Section,
        '$(vStart_$(timerName))'	AS StartTime,
        '$(vEnd_$(timerName))'		AS EndTime,
        '$(vDuration_$(timerName))' AS Duration
    AUTOGENERATE 1;
    
    
END SUB
