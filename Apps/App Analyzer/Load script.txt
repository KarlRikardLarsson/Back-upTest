///$tab Range function
Sub ExtractRange(Field,Table)

	ExtractRange1:
	Load Distinct
		$(Field)_tmp,
		Subfield($(Field)_tmp,'-',1)											as RangeStart,
		Coalesce(Subfield($(Field)_tmp,'-',2),Subfield($(Field)_tmp,'-',1))		as RangeEnd
	Resident $(Table)
	where not isnull($(Field)_tmp);

	ExtractRange2: //Vi löser upp de intervall som är siffor
	Load
		$(Field)_tmp,
		RangeStart+iterno()-1				as $(Field)
	Resident ExtractRange1 
	while RangeStart+iterno()-1 <= RangeEnd and isnum(RangeStart) and isnum(RangeEnd);

	Concatenate(ExtractRange2)
	Load
		$(Field)_tmp,
		RangeStart							as $(Field)
	Resident ExtractRange1 
	where not isnum(RangeStart) or not isnum(RangeEnd);

	Drop Table ExtractRange1;

	Join ($(Table)) Load * Resident ExtractRange2;
	Drop Field $(Field)_tmp;
	Drop Table ExtractRange2;
EndSub

///$tab Reports
Set vConcatenate = ;

IncomeStatement_tmp:
Load * INLINE [Report Name];

For Each vReport in 'Resultaträkning', 'Balansräkning'	//All reports that you are using

	Concatenate (IncomeStatement_tmp)
    Load
    	*,
        "Report Header"																			as "$(vReport)",
    	"Report Header Text"																	as "$(vReport) Text"
        ;
	LOAD
        '$(vReport)'																			as "Report Name",
        Dual(Text(Coalesce("Report Header",'')),num(RecNo()+1))									as "Report Header",
        Coalesce("Report Header",Repeat(' ',num#(RecNo()+1)))									as "Report Header Text",
        RecNo()+1																				as ReportHeaderExcelRow,
        Pick(Match("Calculation type",	'Blank','Sum','SumNoFraction','Division'),
                                        'Blank','Sum','SumNoFraction','Nominator')				as Expression,
        If("Calculation type"='Blank',1,0)														as flagBlankRow,
        "Text color",
        text("Background color")																as "Background color",
        Text(Numberformat)																		as Numberformat,
        "Account Group Type",
        "Account Group level",
        "Account Group Name",
        "Account Group Type"&'|Account Group '&"Account Group level"&'|'&"Account Group Name"	as %ReportHeader,
        ChgColor
    FROM $(vFilePath) 
    (ooxml, embedded labels, table is [$(vReport)])
    where len("Calculation type")>0;
    
    Concatenate (IncomeStatement_tmp)
	Load
	    "Report Header"                                                                             as "$(vReport)",
	    "Report Header Text"                                                                        as "$(vReport) Text",
	    *,
	    "Account Group Type"&'|Account Group '&"Account Group level"&'|'&"Account Group Name"      as %ReportHeader
	    ;
	LOAD
	    '$(vReport)'                                                                                as "Report Name",
	    Dual(Text(Coalesce("Report Header",'')),num(RecNo()+1))                                     as "Report Header",
	    Coalesce("Report Header",Repeat(' ',num#(RecNo()+1)))                                       as "Report Header Text",
	    RecNo()+1                                                                                   as ReportHeaderExcelRow,
	    'Denominator'                                                                               as Expression,
	    0                                                                                           as flagBlankRow,
	    "Text color",
	    "Background color",
	    Text(Numberformat)                                                                          as Numberformat,
	    "Divide by Account Group Type"                                                              as "Account Group Type",
	    "Divide By Account Group Level"                                                             as "Account Group level",
	    "Divide By Account Group Name"                                                              as "Account Group Name",
	    ChgColor
	FROM $(vFilePath)
	(ooxml, embedded labels, table is [$(vReport)])
	where "Calculation type"='Division';

Next
